            } else {
                nextQuestionBtn.style.display = 'none'; // Hide next button on last question
                setTimeout(showResult, 1000); // Show result after a short delay
            }
        }

        function nextQuestion() {
            currentQuestionIndex++;
            renderQuestion();
        }

        function showResult() {
            questionDisplay.innerHTML = `
                <p class="text-2xl font-bold text-susBlue mb-4">Questionário Concluído!</p>
                <p class="text-xl">Você acertou ${score} de ${questions.length} perguntas.</p>
            `;
            quizFeedback.textContent = ''; // Clear feedback
            quizScore.textContent = ''; // Clear score during result display
            nextQuestionBtn.style.display = 'none';
            restartQuizBtn.style.display = 'block';
        }

        function restartQuiz() {
            currentQuestionIndex = 0;
            score = 0;
            restartQuizBtn.style.display = 'none';
            renderQuestion();
        }

        // Event Listeners for control buttons
        nextQuestionBtn.addEventListener('click', nextQuestion);
        restartQuizBtn.addEventListener('click', restartQuiz);

        // Initial render
        renderQuestion();
    </script>
